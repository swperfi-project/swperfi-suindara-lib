 23class LoggerSetup:
 24    @staticmethod
 25    def setup_logger(name: str, log_dir: str = "logs") -> logging.Logger:
 26        """
 27        Sets up and returns a logger with a specific format.
 28        
 29        Parameters:
 30            name (str): The name of the logger.
 31            log_dir (str): Directory where log files will be saved. Default is "logs".
 32        
 33        Returns:
 34            logging.Logger: Configured logger instance.
 35        """
 36        # Create the log directory if it doesn't exist
 37        if not os.path.exists(log_dir):
 38            os.makedirs(log_dir)
 39        
 40        # Create the log file name based on the class/module name
 41        log_file = os.path.join(log_dir, f"{name}.log")
 42
 43        # Set up the logger
 44        logger = logging.getLogger(name)
 45        logger.setLevel(logging.INFO)
 46        
 47        # Check if the logger already has handlers, to avoid adding multiple handlers
 48        if not logger.handlers:
 49            # Create a console handler with the desired format
 50            ch = logging.StreamHandler()
 51            formatter = logging.Formatter("%(asctime)s - %(levelname)s - %(message)s")
 52            ch.setFormatter(formatter)
 53            logger.addHandler(ch)
 54            
 55            # Create a file handler for saving logs to a file
 56            fh = logging.FileHandler(log_file)
 57            fh.setFormatter(formatter)
 58            logger.addHandler(fh)
 59
 60        return logger
