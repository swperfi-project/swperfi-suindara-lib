This module implements an object-oriented data processor for call drop logs. It integrates log parsing (via a LogParser attribute) with further cleaning, transformation, and consolidation of data.  66    def run_processing(self):
 67        """
 68        Runs the complete processing pipeline.
 69        
 70        It checks if the LogParser has raw logs loaded (via its get_logs() method or history).
 71        If raw data exists, it then loads the required partial DataFrames and runs consolidation.
 72        Otherwise, the process is aborted and a status is recorded.
 73        """
 74        self.logger.info("[RUN_PROCESSING] Starting DataProcessor pipeline.")
 75        # Check raw data from LogParser
 76        raw_logs = self.log_parser.get_logs()
 77        if raw_logs.empty:
 78            msg = "LogParser raw logs are empty. Aborting processing."
 79            self.history['raw_logs'] = {'status': msg, 'rows': 0}
 80            self.logger.error("[RUN_PROCESSING] %s", msg)
 81            return
 82        self.logger.info(f"[RUN_PROCESSING] Raw logs loaded successfully. Number of rows: {raw_logs.shape[0]}")
 83        
 84        # Otherwise, proceed with partial DataFrame extraction.
 85        self.logger.info("[RUN_PROCESSING] Invoking load_partial_dataframes().")
 86        self.load_partial_dataframes()
 87        self.logger.info("[RUN_PROCESSING] Invoking consolidate_call_data().")
 88        self.consolidate_call_data()

172    def parse_call_id_status_timestamp(self, df: pd.DataFrame) -> pd.DataFrame:
173        """
174        Extracts call IDs, state transitions, and timestamps.
175        
176        Uses regex patterns from the configuration.
177        """
178        self.logger.info("[CALL_ID_STATUS] Extracting call state transitions.")
179        # Validate required columns
180        for col in ['Log_Tag', 'Tag_Values', 'Timestamp']:
181            if col not in df.columns:
182                raise ValueError(f"Input DataFrame must contain '{col}'.")
183        filtered_df = df[df['Log_Tag'] == 'Telephony:'][['Timestamp', 'Tag_Values']].copy()
184        filtered_df['call_id'] = filtered_df['Tag_Values'].str.extract(CALL_STATE_PATTERNS['call_id'])
185        filtered_df['from_state'] = filtered_df['Tag_Values'].str.extract(CALL_STATE_PATTERNS['from_state'])
186        filtered_df['to_state'] = filtered_df['Tag_Values'].str.extract(CALL_STATE_PATTERNS['to_state'])
187        result_df = filtered_df.dropna(subset=['call_id', 'from_state', 'to_state'])
188        self.logger.info("[CALL_ID_STATUS] Extracted %d records.", len(result_df))
189        return result_df

191    def parse_call_disconnect_causes(self, df: pd.DataFrame) -> pd.DataFrame:
192        """
193        Parses disconnect causes from call logs.
194        """
195        self.logger.info("[DISCONNECT_CAUSES] Parsing disconnect causes.")
196        disconnect_logs = df[(df['Log_Tag'] == 'Telephony:') &
197                             (df['Tag_Values'].str.contains('onDisconnect'))]
198        parsed_data = []
199        for _, row in disconnect_logs.iterrows():
200            log_line = row['Tag_Values']
201            timestamp = row['Timestamp']
202            call_id_match = re.search(DISCONNECT_PATTERNS['call_id'], log_line)
203            cause_match = re.search(DISCONNECT_PATTERNS['cause'], log_line)
204            parsed_data.append({
205                'Timestamp': timestamp,
206                'call_id': call_id_match.group(1) if call_id_match else None,
207                'cause': cause_match.group(1) if cause_match else None
208            })
209        result_df = pd.DataFrame(parsed_data).dropna(subset=['call_id', 'cause'])
210        self.logger.info("[DISCONNECT_CAUSES] Parsed %d disconnect records.", len(result_df))
211        return result_df

289    def parse_signal_strength_df(self, df: pd.DataFrame) -> pd.DataFrame:
290        """
291        Parses signal strength logs and identifies the dominant technology.
292        """
293        self.logger.info("[SIGNAL_STRENGTH] Parsing signal strength logs.")
294        signal_logs = df[df['Tag_Values'].str.contains('SIGNAL_STRENGTH SignalStrength:')]
295        parsed_data = []
296        
297        for _, row in signal_logs.iterrows():
298            log_line = row['Tag_Values']
299            timestamp = row['Timestamp']
300            log_data = {'Timestamp': timestamp}
301
302            # Extract SIM slot
303            sim_slot_match = re.search(r'\[PHONE(\d+)\]', log_line)
304            log_data['PhoneSlot'] = int(sim_slot_match.group(1)) if sim_slot_match else None
305
306            # Extract values for each RAT using imported patterns
307            for rat, rat_patterns in SIGNAL_STRENGTH_PATTERNS.items():
308                for param, pattern in rat_patterns.items():
309                    match = re.search(pattern, log_line)
310                    if match:
311                        log_data[f"{rat}_{param}"] = int(match.group(1))
312
313            parsed_data.append(log_data)
314
315        # Convert to DataFrame
316        parsed_df = pd.DataFrame(parsed_data)
317        # Identify dominant technology per record
318        def identify_dominant_tech(row):
319            INVALID_VALUE = 2147483647
320            for col in row.index:
321                if "_" in col:
322                    tech, _ = col.split("_", 1)
323                    value = row[col]
324                    if value != INVALID_VALUE and pd.notna(value):
325                        return tech
326            return None
327        parsed_df['Dominant_Technology'] = parsed_df.apply(identify_dominant_tech, axis=1)
328        self.logger.info("[SIGNAL_STRENGTH] Parsed %d records.", len(parsed_df))
329        return parsed_df

368        """

379        self.logger.debug(f"[{TAG}] Found {len(filtered_df)} records after filtering.")

382            return filtered_df.sort_values(by='Timestamp', ascending=False).iloc[0].to_dict()
383        else:
384            self.logger.warning(f"[{TAG}] No records found; returning None for all fields.")

387    def get_signal_strength_record(self, signal_df: pd.DataFrame, start_time: datetime, disc_time: datetime, phone_slot) -> dict:
388        """
389        Processes signal strength records for a given phone slot and call interval.
390        Calculates the mean of signal metrics within the interval or retrieves the closest record.
391        """
392        TAG = "SIGNAL_STRENGTH"
393        self.logger.debug(f"[{TAG}] Processing signal strength for PhoneSlot={phone_slot}, Start={start_time}, Disc={disc_time}.")
394
395        # Garantindo que phone_slot seja um valor escalar
396        phone_slot = phone_slot if isinstance(phone_slot, (int, str)) else phone_slot.iloc[0]
397
398        # Filtrar registros por PhoneSlot
399        filtered_signals = signal_df[signal_df['PhoneSlot'] == phone_slot].dropna(subset=['Dominant_Technology'])
400        
401        # Verificar se h registros aps filtro de PhoneSlot
402        if filtered_signals.empty:
403            self.logger.warning(f"[{TAG}] No signals found for PhoneSlot={phone_slot}. Returning empty record.")
404            return {col: None for col in signal_df.columns}
405
406        # Garantindo que a coluna Timestamp seja datetime
407        try:
408            filtered_signals['Timestamp'] = pd.to_datetime(filtered_signals['Timestamp'])
409        except Exception as e:
410            self.logger.error(f"[{TAG}] Error parsing timestamps: {e}")
411            return {col: None for col in signal_df.columns}
412
413        self.logger.debug(f"[{TAG}] Found {len(filtered_signals)} records for PhoneSlot {phone_slot}.")
414
415        # Filtrar registros dentro do intervalo da chamada
416        valid_signals = filtered_signals[
417            (filtered_signals['Timestamp'] >= start_time) & 
418            (filtered_signals['Timestamp'] <= disc_time)
419        ]
420
421        self.logger.debug(f"[{TAG}] Found {len(valid_signals)} records within the call interval.")
422
423        if not valid_signals.empty:
424            numeric_columns = valid_signals.select_dtypes(include=['number']).columns
425            averaged_row = valid_signals[numeric_columns].mean(axis=0).apply(int).to_dict()
426            averaged_row['Timestamp'] = valid_signals['Timestamp'].max()
427            averaged_row['PhoneSlot'] = phone_slot
428            averaged_row['Dominant_Technology'] = valid_signals['Dominant_Technology'].mode()[0]
429            self.logger.debug(f"[{TAG}] Calculated interval metrics: {averaged_row}")
430            return averaged_row
431
432        # Caso no existam registros no intervalo, buscar registros mais prximos
433        before_start = filtered_signals[filtered_signals['Timestamp'] < start_time].sort_values(by='Timestamp', ascending=False)
434        after_end = filtered_signals[filtered_signals['Timestamp'] > disc_time].sort_values(by='Timestamp', ascending=True)
435
436        self.logger.debug(f"[{TAG}] Records before start: {len(before_start)}, after end: {len(after_end)}.")
437
438        # Escolher o registro mais prximo
439        closest_record = None
440        if not after_end.empty:
441            closest_record = after_end.iloc[0]
442            self.logger.debug(f"[{TAG}] Using record after disconnect: {closest_record['Timestamp']}.")
443        elif not before_start.empty:
444            closest_record = before_start.iloc[0]
445            self.logger.debug(f"[{TAG}] Using record before start: {closest_record['Timestamp']}.")
446
447        if closest_record is not None:
448            record_dict = closest_record.to_dict()
449            self.logger.debug(f"[{TAG}] Returning closest record: {record_dict}")
450            return record_dict
451
452        # Se nenhum registro vlido for encontrado, retornar um dicionrio vazio
453        self.logger.warning(f"[{TAG}] No valid signal records found. Returning empty record.")
454        return {col: None for col in signal_df.columns}

585    def consolidate_call_data(self, tolerance: pd.Timedelta = pd.Timedelta(seconds=30)) -> None:
586        """
587        Consolidates call data by merging the partial DataFrames stored in self.required_dfs.
588        Description:
589        ------------
590        This function combines parsed data from different logs (e.g., call status, signal strength, 
591        registration state, and persist storage) to create a comprehensive DataFrame for each call.
592
593        Parameters:
594        -----------
595        df_raw : DataFrame
596            The raw DataFrame containing all log data.
597        tolerance : pd.Timedelta, optional
598            Time tolerance for associating PersistAtomsStorage records with calls, by default 30 seconds.
599
600        Returns:
601        --------
602        DataFrame
603            A consolidated DataFrame with detailed information about each call, including:
604            - Call start, active, and disconnect times.
605            - Disconnect causes and persist storage metrics.
606            - Signal strength statistics and registration state.
607            - Dominant technology and other relevant parameters.
608
609        Workflow:
610        ---------
611        1. **Extract Call Status**:
612            - Parses call start (`IDLE` to `ACTIVE`), activation, and disconnect times using `get_call_id_status_timestamp`.
613
614        2. **Parse Disconnect Causes**:
615            - Extracts disconnect reasons using `parse_call_disconnect_causes`.
616
617        3. **Process PersistAtomsStorage Logs**:
618            - Extracts metrics such as signal strength and RAT transitions using `parse_persist_atoms_storage_calls`.
619
620        4. **Merge Registration State Information**:
621            - Retrieves the most recent registration state (`parse_voice_registration_state`) for each call.
622
623        5. **Integrate Signal Strength Data**:
624            - Uses `get_signal_strength_record` to calculate signal metrics for the call interval or fetch nearby records.
625
626        6. **Combine Data into a Unified Format**:
627            - Merges all processed data into a single DataFrame, ensuring timestamps and `call_id` align correctly.
628
629        Notes:
630        ------
631        - This function acts as the final step in the pipeline, bringing together multiple log sources.
632        - It ensures no data is duplicated and prioritizes the most relevant records for each call.
633
634        Example:
635        --------
636        consolidated_data = consolidate_call_data(df_raw, tolerance=pd.Timedelta(seconds=30))
637
638        Output:
639        -------
640        | call_id     | start_time          | act_time            | disc_time          | cause | signal_strength | ...
641        |-------------|---------------------|---------------------|--------------------|-------|-----------------|---
642        | TC@1234_1   | 2024-11-29 10:00:00 | 2024-11-29 10:01:00 | 2024-11-29 10:05:00 | DROP  | {'LTE_rsrp': -90, ...}
643        """
644        self.logger.info("[CONSOLIDATION] Starting consolidation process.")
645        
646        try:
647            #  Verificao inicial dos DataFrames
648            self.logger.info("[CONSOLIDATION] Checking required DataFrames.")
649            for key, df in self.required_dfs.items():
650                self.logger.info(f"[CONSOLIDATION] [{key.upper()}] {len(df)} rows, columns: {list(df.columns)}")
651                if df.empty:
652                    msg = f"Empty after processing: {key} is empty - consolidation terminated."
653                    self.history['consolidation'] = {'status': msg, 'call_count_before': 0, 'call_count_after': 0}
654                    self.logger.warning("[CONSOLIDATION] %s", msg)
655                    self.consolidated_df = pd.DataFrame()
656                    return
657
658
659            #  Extrao dos DataFrames parciais necessrios
660            status_df = self.required_dfs['status_df']
661            disconnect_df = self.required_dfs['disconnect_df']
662            persist_atoms_df = self.required_dfs['persist_atoms_df']
663            voice_reg_df = self.required_dfs['voice_reg_df']
664            signal_df = self.required_dfs['signal_df']
665            ims_tracker_df = self.required_dfs['ims_tracker_df']
666
667            
668
669
670            #  Consolidao dos tempos de chamada
671            self.logger.info("[CONSOLIDATION] [CALL STATUS] Consolidating call times.")
672            call_ends = status_df[status_df['to_state'] == 'DISCONNECTED'].rename(columns={'Timestamp': 'disc_time'})
673            call_ends = call_ends.sort_values(by='disc_time').drop_duplicates(subset=['call_id'], keep='first')
674
675            call_active = status_df[status_df['to_state'] == 'ACTIVE'].rename(columns={'Timestamp': 'act_time'})
676            call_active = call_active.sort_values(by='act_time').drop_duplicates(subset=['call_id'], keep='first')
677
678            call_starts = status_df[status_df['from_state'] == 'IDLE'].rename(columns={'Timestamp': 'start_time'})
679            call_starts = call_starts.sort_values(by='start_time').drop_duplicates(subset=['call_id'], keep='first')
680
681            calls_df = call_ends[['call_id', 'disc_time']].merge(
682                call_active[['call_id', 'act_time']], on='call_id', how='left'
683            ).merge(
684                call_starts[['call_id', 'start_time']], on='call_id', how='left'
685            )
686            self.logger.info(f"[CONSOLIDATION] [CALL STATUS] Calls DataFrame after consolidation: {calls_df.shape}")
687
688            # Merge de disconnect_df com calls_df
689            self.logger.info("[CONSOLIDATION] [DISCONNECT] Merging disconnect causes.")
690            calls_df = calls_df.merge(disconnect_df, on='call_id', how='left')
691            
692
693
694            #  Integrao de logs do IMSPhone Tracker
695            self.logger.info("[CONSOLIDATION] [IMS TRACKER] Merging IMSPhone Tracker logs.")
696            ims_tracker_df['Timestamp'] = pd.to_datetime(ims_tracker_df['Timestamp'])
697            ims_tracker_selected = ims_tracker_df[['call_id', 'ImsTracker_cause', 'wifi_st', 'ims_networkType']].drop_duplicates()
698
699            calls_df = calls_df.merge(ims_tracker_selected, on='call_id', how='inner', validate="one_to_one")
700            self.logger.info(f"[CONSOLIDATION] [IMS TRACKER] Calls DataFrame after merge: {calls_df.shape}")
701
702            #  Adio de informaes do PersistAtomsStorage
703            self.logger.info("[CONSOLIDATION] [PERSIST ATOMS] Merging PersistAtomsStorage data.")
704            persist_atoms_df['Timestamp'] = pd.to_datetime(persist_atoms_df['Timestamp'])
705            persist_atoms_df = persist_atoms_df.rename(columns={'Timestamp': 'persist_time'})
706
707            persist_atoms_merged = calls_df.apply(
708                lambda row: persist_atoms_df[
709                    (persist_atoms_df['persist_time'] <= row['disc_time'] + tolerance) &
710                    (persist_atoms_df['persist_time'] >= (row['act_time'] if pd.notna(row['act_time']) else row['start_time']))
711                ].sort_values(by='persist_time', ascending=False).head(1),
712                axis=1
713            )
714            persist_atoms_expanded = pd.concat(persist_atoms_merged.tolist(), ignore_index=True)
715            calls_df = pd.concat([calls_df, persist_atoms_expanded], axis=1)
716            self.logger.info(f"[CONSOLIDATION] [PERSIST ATOMS] Calls DataFrame after merge: {calls_df.shape}")
717            
718            #  Adio de informaes de Registro de Voz
719            self.logger.info("[CONSOLIDATION] [VOICE REGISTRATION] Merging Voice Registration data.")
720            voice_reg_df['Timestamp'] = pd.to_datetime(voice_reg_df['Timestamp'])
721            voice_reg_merged = calls_df.apply(
722                lambda row: self.get_most_recent_record(
723                    voice_reg_df,
724                    cut_time=row['disc_time'] + tolerance,
725                    phone_slot=row.get('PhoneSlot'),
726                    TAG="VOICE_REGISTRATION"
727                ),
728                axis=1
729            )
730            voice_reg_expanded = pd.DataFrame(voice_reg_merged.tolist())
731            if 'Timestamp' in voice_reg_expanded.columns:
732                voice_reg_expanded = voice_reg_expanded.rename(columns={'Timestamp': 'voice_reg_time'})
733            calls_df = pd.concat([calls_df, voice_reg_expanded], axis=1)
734            self.logger.info(f"[CONSOLIDATION] [VOICE REGISTRATION] Calls DataFrame after merge: {calls_df.shape}")
735
736            #  Integrao de dados de Fora de Sinal
737            self.logger.info("[CONSOLIDATION] [SIGNAL STRENGTH] Merging Signal Strength data.")
738            signal_df['Timestamp'] = pd.to_datetime(signal_df['Timestamp'])
739            signal_strength_merged = calls_df.apply(
740                lambda row: self.get_signal_strength_record(
741                    signal_df,
742                    (row['act_time'] if pd.notna(row['act_time']) else row['start_time']) - tolerance, row['disc_time'] + tolerance,
743                    row.get('PhoneSlot')
744                ),
745                axis=1
746            )
747            signal_strength_expanded = pd.DataFrame(signal_strength_merged.tolist())
748            if 'Timestamp' in signal_strength_expanded.columns:
749                signal_strength_expanded = signal_strength_expanded.rename(columns={'Timestamp': 'signal_strength_time'})
750
751            calls_df = pd.concat([calls_df, signal_strength_expanded], axis=1)
752
753            # Garantir que PhoneSlot no seja duplicado
754            calls_df = calls_df.loc[:, ~calls_df.columns.duplicated()]  # Remover colunas duplicadas
755
756            # Adicionar colunas de fora de sinal relevantes ao DataFrame final
757            signal_columns = [col for col in signal_strength_expanded.columns if col not in {'Timestamp', 'PhoneSlot'}] # 'Dominant_Technology'
758            
759
760             #  **Seleo das colunas relevantes**
761            self.logger.info("[CONSOLIDATION] Selecting relevant columns.")
762            self.consolidated_df = calls_df[RELEVANT_COLUMNS + signal_columns]  # Adicionar colunas de fora de sinal
763
764
765            self.consolidated_df = self.filter_signal_strength_by_dominant_tech(self.consolidated_df)
766
767            # Valor padro para imputao
768            default_value = "2147483647"
769
770
771            # Para cada tecnologia, verificar quais colunas existem no DataFrame e aplicar fillna
772            for tech, cols in FILTER_RAT_COLUMNS.items():
773                existing_cols = [col for col in cols if col in self.consolidated_df.columns]
774                if existing_cols:
775                    self.consolidated_df[existing_cols] = self.consolidated_df[existing_cols].fillna(default_value)
776
777
778            #  **Verificao Final**
779            call_count_before = len(status_df)
780            call_count_after = len(self.consolidated_df)
781
782            # Aplicar a classificao do status da chamada
783            self.logger.info("[CONSOLIDATION] [IDTAG] Apply the classification call status based on `disconnect cause`.")
784            if not self.consolidated_df.empty:
785                self.consolidated_df['IDTAG'] = self.consolidated_df.apply(self.classify_call_status, axis=1)
786                missing_call_ids = set(disconnect_df['call_id']) - set(self.consolidated_df['call_id'])
787                if missing_call_ids:
788                    status_message = f"Success with warnings - Missing call events: {', '.join(map(str, missing_call_ids))}."
789                else:
790                    status_message = "Success - Consolidation complete."
791            else:
792                status_message = "Consolidation failed - consolidated_df is empty."
793                self.logger.warning("[CONSOLIDATION] %s", status_message)
794                self.consolidated_df['IDTAG'] = None
795
796            # Incorporar informaes do report_info
797            if self.log_parser.report_info:
798                self.logger.info("[CONSOLIDATION] Incorporating report information.")
799
800                build_id = self.log_parser.report_info.get('build_id', '')
801                if build_id:
802                    android_version = self.get_android_version_from_build_id(build_id)
803                    build_code = self.generate_unique_code_4_bldid(build_id)
804                    self.consolidated_df['android_version'] = android_version
805                    self.consolidated_df['build_code'] = build_code
806                    self.logger.info(f"[CONSOLIDATION] Build ID: {build_id} | Android Version: {android_version} | Build Code: {build_code}")
807
808                product_name = self.log_parser.report_info.get('product', '')
809                if product_name:
810                    product_code = self.generate_unique_code(product_name)
811                    self.consolidated_df['product_code'] = product_code
812                    self.logger.info(f"[CONSOLIDATION] Product Name: {product_name} | Product Code: {product_code}")
813
814                # Aplicar rotulagem de chamadas drop com base nas informaes do report
815                self.logger.info("[CONSOLIDATION] Labeling dropped calls based on report info.")
816                self.consolidated_df = self.label_is_drop_from_dict(self.consolidated_df, self.log_parser.report_info)
817
818                # Resumo final da incorporao
819                self.logger.info(f"[CONSOLIDATION] Report info successfully incorporated. {len(self.consolidated_df)} rows updated.")
820            else:
821                self.logger.warning("[CONSOLIDATION] No report information available. Skipping incorporation.")
822            
823            #  Criao de colunas derivadas
824            self.logger.info("[CONSOLIDATION] Creating additional time-based columns.")
825            self.consolidated_df['disc_time'] = pd.to_datetime(self.consolidated_df['disc_time'], errors='coerce')
826            self.consolidated_df['day_of_week'] = self.consolidated_df['disc_time'].dt.dayofweek.astype(int) + 1
827            self.consolidated_df['hour_of_day'] = self.consolidated_df['disc_time'].dt.hour.astype(int)
828
829
830            #  **Atualizao do Histrico**
831            self.history['consolidation'] = {
832                'status': status_message,
833                'call_count_before': call_count_before,
834                'call_count_after': call_count_after
835            }
836            self.logger.info("[CONSOLIDATION] %s", status_message)
837
838        except Exception as e:
839            status_message = f"Error during consolidation: {e}"
840            self.logger.error("[CONSOLIDATION] %s", status_message)
841            self.consolidated_df = pd.DataFrame()

388    def get_signal_strength_record(self, signal_df: pd.DataFrame, start_time: datetime, disc_time: datetime, phone_slot) -> dict:
389        """
390        Processes signal strength records for a given phone slot and call interval.
391        Calculates the mean of signal metrics within the interval or retrieves the closest record.
392        """
393        TAG = "SIGNAL_STRENGTH"
394        self.logger.debug(f"[{TAG}] Processing signal strength for PhoneSlot={phone_slot}, Start={start_time}, Disc={disc_time}.")
395
396        # Garantindo que phone_slot seja um valor escalar
397        phone_slot = phone_slot if isinstance(phone_slot, (int, str)) else phone_slot.iloc[0]
398
399        # Filtrar registros por PhoneSlot
400        filtered_signals = signal_df[signal_df['PhoneSlot'] == phone_slot].dropna(subset=['Dominant_Technology'])
401        
402        # Verificar se h registros aps filtro de PhoneSlot
403        if filtered_signals.empty:
404            self.logger.warning(f"[{TAG}] No signals found for PhoneSlot={phone_slot}. Returning empty record.")
405            return {col: None for col in signal_df.columns}
406
407        # Garantindo que a coluna Timestamp seja datetime
408        try:
409            filtered_signals['Timestamp'] = pd.to_datetime(filtered_signals['Timestamp'])
410        except Exception as e:
411            self.logger.error(f"[{TAG}] Error parsing timestamps: {e}")
412            return {col: None for col in signal_df.columns}
413
414        self.logger.debug(f"[{TAG}] Found {len(filtered_signals)} records for PhoneSlot {phone_slot}.")
415
416        # Filtrar registros dentro do intervalo da chamada
417        valid_signals = filtered_signals[
418            (filtered_signals['Timestamp'] >= start_time) & 
419            (filtered_signals['Timestamp'] <= disc_time)
420        ]
421
422        self.logger.debug(f"[{TAG}] Found {len(valid_signals)} records within the call interval.")
423
424        if not valid_signals.empty:
425            numeric_columns = valid_signals.select_dtypes(include=['number']).columns
426            averaged_row = valid_signals[numeric_columns].mean(axis=0).apply(int).to_dict()
427            averaged_row['Timestamp'] = valid_signals['Timestamp'].max()
428            averaged_row['PhoneSlot'] = phone_slot
429            averaged_row['Dominant_Technology'] = valid_signals['Dominant_Technology'].mode()[0]
430            self.logger.debug(f"[{TAG}] Calculated interval metrics: {averaged_row}")
431            return averaged_row
432
433        # Caso no existam registros no intervalo, buscar registros mais prximos
434        before_start = filtered_signals[filtered_signals['Timestamp'] < start_time].sort_values(by='Timestamp', ascending=False)
435        after_end = filtered_signals[filtered_signals['Timestamp'] > disc_time].sort_values(by='Timestamp', ascending=True)
436
437        self.logger.debug(f"[{TAG}] Records before start: {len(before_start)}, after end: {len(after_end)}.")
438
439        # Escolher o registro mais prximo
440        closest_record = None
441        if not after_end.empty:
442            closest_record = after_end.iloc[0]
443            self.logger.debug(f"[{TAG}] Using record after disconnect: {closest_record['Timestamp']}.")
444        elif not before_start.empty:
445            closest_record = before_start.iloc[0]
446            self.logger.debug(f"[{TAG}] Using record before start: {closest_record['Timestamp']}.")
447
448        if closest_record is not None:
449            record_dict = closest_record.to_dict()
450            self.logger.debug(f"[{TAG}] Returning closest record: {record_dict}")
451            return record_dict
452
453        # Se nenhum registro vlido for encontrado, retornar um dicionrio vazio
454        self.logger.warning(f"[{TAG}] No valid signal records found. Returning empty record.")
455        return {col: None for col in signal_df.columns}

605            - Call start, active, and disconnect times.

607            - Signal strength statistics and registration state.
608            - Dominant technology and other relevant parameters.
609

611        ---------

613            - Parses call start (`IDLE` to `ACTIVE`), activation, and disconnect times using `get_call_id_status_timestamp`.

  • This function acts as the final step in the pipeline, bringing together multiple log sources.
